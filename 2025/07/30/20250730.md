BEGIN {
    string input = "3 2 1 2 2 3 6 1";
    int i, count;
    string parts[];
    string cur = input;
    string sep = " ";
    int seplen = length(sep);

    count = 0;
    while (match(cur, sep) != -1) {
        parts[count] = substr(cur, 0, match(cur, " "));
        cur = substr(cur, match(cur, " ") + seplen);
        count++;
    }
    parts[count] = cur;
    print (# parts);
    for (parts[i]) {
        printf("i: %s, parts[i]: %s\n", i, parts[i]);
    }
}
